001package org.grails.plugin.resources.stash;
002
003import org.grails.plugin.resource.util.DispositionsUtils;
004
005import javax.servlet.http.HttpServletRequest;
006import java.io.IOException;
007import java.io.Writer;
008import java.util.*;
009
010/**
011 * Manages the stashing and unstashing of page fragments.
012 *
013 * @author Patrick Jungermann
014 */
015public class StashManager {
016
017    /**
018     * Prefix used for storing page fragment stashes.
019     */
020    public static final String REQ_ATTR_PREFIX_PAGE_FRAGMENTS = "resources.plugin.page.fragments";
021
022    /**
023     * Registered, usable stash writers.
024     */
025    public static final Map<String, StashWriter> STASH_WRITERS = new HashMap<String, StashWriter>();
026    static {
027        // register the basic stash writers
028        STASH_WRITERS.put("script", new ScriptStashWriter());
029        STASH_WRITERS.put("style", new StyleStashWriter());
030    }
031
032    /**
033     * Stashes a page fragment.
034     *
035     * @param request        The current request, at which the page fragment has to be stashed.
036     * @param type           The stash's (writer) type.
037     * @param disposition    The disposition, at which the page fragment has to be unstashed.
038     * @param fragment       The fragment, which has to be stashed.
039     */
040    @SuppressWarnings("unchecked")
041    public static void stashPageFragment(final HttpServletRequest request, final String type, final String disposition, final String fragment) {
042        final String resourceTrackerName = makePageFragmentKey(type, disposition);
043        DispositionsUtils.addDispositionToRequest(request, disposition, "-page-fragments-");
044
045        List<String> resourceTracker = (List<String>) request.getAttribute(resourceTrackerName);
046        if (resourceTracker == null) {
047            resourceTracker = new ArrayList<String>();
048            request.setAttribute(resourceTrackerName, resourceTracker);
049        }
050        resourceTracker.add(fragment);
051    }
052
053    /**
054     * Unstashes the disposition's page fragments.
055     *
056     * @param out            The target, to which all fragments have to be written to.
057     * @param request        The request, at which the fragments have been stashed.
058     * @param disposition    The disposition, for which all fragments have to be rendered.
059     * @throws IOException if there was any problem with writing the fragments.
060     */
061    public static void unstashPageFragments(final Writer out, final HttpServletRequest request, final String disposition) throws IOException {
062        for (final String type : STASH_WRITERS.keySet()) {
063            List<String> stash = consumePageFragments(request, type, disposition);
064            if (!stash.isEmpty()) {
065                STASH_WRITERS.get(type).write(out, stash);
066            }
067        }
068    }
069
070    /**
071     * Returns the stash (all page fragments) of the requested type and disposition.
072     *
073     * @param request        The request, at which the fragments have been stashed.
074     * @param type           The fragments' type.
075     * @param disposition    The disposition, for which the fragments have to be returned.
076     * @return All fragments of the requested type and disposition.
077     */
078    @SuppressWarnings("unchecked")
079    private static List<String> consumePageFragments(final HttpServletRequest request, final String type, final String disposition) {
080        List<String> stash = (List<String>) request.getAttribute(makePageFragmentKey(type, disposition));
081        return stash != null ? stash : Collections.EMPTY_LIST;
082    }
083
084    /**
085     * Returns the page fragment key, used to store the page fragment stashes.
086     *
087     * @param type           The fragments' type.
088     * @param disposition    The fragments' disposition.
089     * @return The page fragment key.
090     */
091    private static String makePageFragmentKey(final String type, final String disposition) {
092        return REQ_ATTR_PREFIX_PAGE_FRAGMENTS + ":" + type + ":" + disposition;
093    }
094
095}
