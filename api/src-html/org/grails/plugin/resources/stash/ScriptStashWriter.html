001package org.grails.plugin.resources.stash;
002
003import java.io.IOException;
004import java.io.Writer;
005import java.util.List;
006
007/**
008 * Writes stashed scripts to the output. Each fragment will be written to separate script tags as an error
009 * inside of one of them would stop the execution of the following ones. This might be harmful, e.g.,
010 * in case of (externally provided and/or dependent) tracking scripts.
011 *
012 * @author Patrick Jungermann
013 */
014public class ScriptStashWriter implements StashWriter {
015
016    @Override
017    public void write(final Writer out, final List<String> stash) throws IOException {
018        for (final String fragment : stash) {
019            out.write("<script type=\"text/javascript\">");
020            out.write(fragment);
021            out.write("</script>");
022        }
023    }
024
025}
